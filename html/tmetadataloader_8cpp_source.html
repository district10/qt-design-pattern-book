00114 void TagLib::MetaDataLoader::checkForWork() {
00115     MetaDataFunctor functor;
00116     if (m_queue.isEmpty() && !m_running) {
00117         m_processingMax = 0;
00118         return;
00119     }
00120     if (m_running ) return;
00121     m_running = true;
00122     m_canceled = false;
00123     while (!m_queue.isEmpty()) {
00124         QStringList sl = m_queue;
00125         m_queue = QStringList();
00126         m_processingMax = sl.length();
00127         emit progressRangeChanged(0, m_processingMax);
00128         for (int i=0; i<m_processingMax;++i) {
00129             if (m_canceled) break;
